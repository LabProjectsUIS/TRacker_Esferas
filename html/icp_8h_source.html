3 /*-------------------------------------------------------
4 Interfaz 2016, all rights reserved.
5 
6 This code defines a set of functions to align rigid objects.
7 Both objects, from now on referred as source and target,
8 are Point Clouds.
9 
10 Registration is performed in two steps. First, the
11 source is brought near the target in a process involving
12 voxelization and a feature-based RANSAC; prior to
13 features estimation, normals are to be computed. Second,
14 an ICP algorithm is applied to source in order to match
15 it to target.
16 
17 Point clouds can be read from text files using read_file.
18 
19 Author: Jesús Muñoz
20 Version: 0.1
21 ---------------------------------------------------------*/

57 void read_file(const std::string& file, int numel, PointCloudT::Ptr cloud) {
58  cloud->width = numel;
59  cloud->height = 1;
60  cloud->is_dense = true;
61  cloud->points.resize(cloud->width * cloud->height);
62 
63  std::ifstream readFile(file);
64 
65  // Load data
66  for (size_t i = 0; i < cloud->points.size(); ++i)
67  {
68  std::string line;
69  std::string var1, var2, var3;
70 
71  getline(readFile, line);
72  std::istringstream ss(line);
73  ss >> var1 >> var2 >> var3;
74  cloud->points[i].x = stof(var1);
75  cloud->points[i].y = stof(var2);
76  cloud->points[i].z = stof(var3);
77  }

86 void bring_cloud_near(PointCloudT::Ptr object, PointCloudT::Ptr scene, Eigen::Matrix4f& transformation, int flag) {
87 
88  FeatureCloudT::Ptr object_features(new FeatureCloudT);
89  FeatureCloudT::Ptr scene_features(new FeatureCloudT);
90  PointCloudT::Ptr object_aligned(new PointCloudT);
91  RadiusOutlierRemovalT outlierRemoval;
92  NormalEstimationT nest;
93  VoxelGridT grid;
94  KdTreeT::Ptr kdtree(new pcl::search::KdTree<PointNT>);
95 
96  const float leaf = 10;
97  double radius = 10;
98 
99  switch (flag) {
100  case 1:
101  // Remove outliers
102  outlierRemoval.setInputCloud(object);
103  // Every point must have 10 neigbors within 10mm, or it will be removed.
104  outlierRemoval.setRadiusSearch(10);
105  outlierRemoval.setMinNeighborsInRadius(10);
106  outlierRemoval.filter(*object);
107  radius = 5;
108  nest.setSearchMethod(kdtree);
109  break;
110  case 2:
111  // Downsample
112  grid.setLeafSize(leaf, leaf, leaf);
113  grid.setInputCloud(object);
114  grid.filter(*object);
115  grid.setInputCloud(scene);
116  grid.filter(*scene);
117  radius = 0.1;
118  break;
119  default:
120  break;
121  }
122 
123  // Estimate normals for scene
124  nest.setRadiusSearch(radius);
125  nest.setInputCloud(scene);
126  nest.compute(*scene);
127 
128  // Estimate features
129  FeatureEstimationT fest;
130  fest.setRadiusSearch(50);
131  fest.setInputCloud(object);
132  fest.setInputNormals(object);
133  fest.compute(*object_features);
134  fest.setInputCloud(scene);
135  fest.setInputNormals(scene);
136  fest.compute(*scene_features);
137 
138  // Perform alignment
139  SampleConsensusPrerejectiveT align;
140  align.setInputSource(object);
141  align.setSourceFeatures(object_features);
142  align.setInputTarget(scene);
143  align.setTargetFeatures(scene_features);
144  align.setMaximumIterations(50000); // Number of RANSAC iterations
145  align.setNumberOfSamples(10); // Number of points to sample for generating/prerejecting a pose
146  align.setCorrespondenceRandomness(10); // Number of nearest features to use
147  align.setSimilarityThreshold(0.0001f); // Polygonal edge length similarity threshold
148  align.setMaxCorrespondenceDistance(1.01f * leaf); // Inlier threshold
149  align.setInlierFraction(0.95f); // Required inlier fraction for accepting a pose hypothesis
150  {
151  pcl::ScopeTime t("Alignment");
152  align.align(*object_aligned);
153  }
154 
155 
156  if (align.hasConverged())
157  {
158  pcl::transformPointCloud(*object, *object, align.getFinalTransformation());
159  transformation = align.getFinalTransformation().inverse();
160  }
161  else
162  {
163  pcl::console::print_error("Alignment failed!\n");
164  }
165 }
