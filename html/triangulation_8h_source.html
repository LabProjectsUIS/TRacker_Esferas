39  *
40  * Function that computes the position of a set on N points given the left and right image projections.
41  * The cameras are assumed to be calibrated, intrinsically, and extrinsically.
42  *
43  * Input:
44  * xL : 2xN matrix of pixel coordinates in the left image
45  * xR : 2xN matrix of pixel coordinates in the right image
46  * om, T : rotation vector and translation vector between right and left cameras(output of stereo calibration)
47  * fc_left, cc_left, ... : intrinsic parameters of the left camera(output of stereo calibration)
48  * fc_right, cc_right, ... : intrinsic parameters of the right camera(output of stereo calibration)
49  *
50  * Output :
51  *
52  * XL : 3xN matrix of coordinates of the points in the left camera reference frame
53  * XR : 3xN matrix of coordinates of the points in the right camera reference frame
54  *
55  * Note : XR and XL are related to each other through the rigid motion equation : XR = R * XL + T, where R = rodrigues(om)

60  void stereo_triangulation(Mat_<double> xL, Mat_<double> xR, Mat_<double> om, Mat_<double> T, Mat_<double> fc_left, Mat_<double> cc_left, vector<double> kc_left, double alpha_c_left, Mat_<double> fc_right, Mat_<double> cc_right, vector<double> kc_right, double alpha_c_right, Mat_<double>& XL, Mat_<double>& XR) {
61  Mat_<double> xt, xtt, R, u, T_vect, DD, n_xt2, n_xtt2, u2, u3, dot_uT, dot_xttT, dot_xttu, NN1, NN2, Zt, Ztt, X1, X2;
62  normalize_pixel(xL, fc_left, cc_left, kc_left, alpha_c_left, xt);
63  normalize_pixel(xR, fc_right, cc_right, kc_right, alpha_c_right, xtt);
64  xt.push_back(cv::Mat::ones(1, xt.cols, CV_64F));
65  xtt.push_back(cv::Mat::ones(1, xtt.cols, CV_64F));
66  int N = xt.cols;
67 
68  //-- - Rotation matrix corresponding to the rigid motion between left and right cameras :
69  Rodrigues(om, R);
70 
71  //-- - Triangulation of the rays in 3D space :
72  u = R * xt;
73  dotProduct(xt, xt, n_xt2);
74  dotProduct(xtt, xtt, n_xtt2);
75 
76  T_vect = repeat(T, 1, N);
77  dotProduct(u, xtt, u2);
78  pow(u2, 2, u2);
79  DD = n_xt2.mul(n_xtt2) - u2;
80 
81  dotProduct(u, T_vect, dot_uT);
82  dotProduct(xtt, T_vect, dot_xttT);
83  dotProduct(u, xtt, dot_xttu);
84 
85  NN1 = dot_xttu.mul(dot_xttT) - n_xtt2.mul(dot_uT);
86  NN2 = n_xt2.mul(dot_xttT) - dot_uT.mul(dot_xttu);
87 
88  divide(NN1.t(), DD.t(), Zt);
89  divide(NN2.t(), DD.t(), Ztt);
90  //cout << Zt << endl << endl;
91  extendM(Zt, xt.cols);
92  extendM(Ztt, xtt.cols);
93 
94  //cout << xt << endl << Zt << endl;
95  X1 = xt.mul(Zt);
96  X2 = R.t()*(xtt.mul(Ztt) - T_vect);
97 
98  //-- - Left coordinates :
99  XL = 0.5 * (X1 + X2);
100  //-- - Right coordinates :
101  XR = R*XL + T_vect;
102  }

158  void comp_distortion_oulu(Mat_<double> xd, vector<double> k, Mat_<double>& x) {
159  double k1 = k[0];
160  double k2 = k[1];
161  double k3 = k[4];
162  double p1 = k[2];
163  double p2 = k[3];
164 
165  xd.copyTo(x); // initial guess
166 
167  for (int i = 0; i < 19; i++) {
168  Mat_<double> x0, x1, x2, x3, x4, r_2;
169  pow(x, 2, x2);
170  reduce(x2, r_2, 0, REDUCE_SUM);
171  pow(r_2, 2, x0);
172  pow(r_2, 3, x1);
173 
174  Mat_<double> k_radial = 1 + k1 * r_2 + k2 * x0 + k3 * x1;
175  pow(x.row(1), 2, x3);
176  pow(x.row(0), 2, x4);
177 
178  Mat_<double> m = 2 * p1*x.row(0).mul(x.row(1)) + p2*(r_2 + 2 * x4);
179  Mat_<double> n = p1 * (r_2 + 2 * x3) + 2 * p2*x.row(0).mul(x.row(1));
180 
181  Mat_<double> delta_x;
182  delta_x.push_back(m);
183  delta_x.push_back(n);
184 
185  Mat_<double> t = (Mat_<double>(2, 1) << 1, 1);
186 
187  divide((xd - delta_x), t*k_radial, x);
188  }
